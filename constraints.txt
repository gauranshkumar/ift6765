# constraints.txt
# Hard version ceilings to prevent Compute Canada dummy-wheel traps AND
# known package incompatibilities.
#
# ── PyArrow / Arrow module ────────────────────────────────────────────────────
# arrow/17.0.0 → pyarrow 17.x; datasets 3.0+ needs pyarrow>=21 → cap it
datasets<=2.19.0

# ── OpenCV module ─────────────────────────────────────────────────────────────
# opencv is managed by the CC custom vllm wheel — do NOT cap it here.
# Load the opencv module before venv activation and install via --no-index;
# the CC vllm wheel will satisfy opencv>=4.11 from that pre-built wheel.

# ── transformers / vllm compatibility ─────────────────────────────────────────
# Qwen3-VL (qwen3_vl architecture) was added in transformers 4.51.0.
# vllm 0.8+ rewrote the tokenizer integration and no longer calls the old
# all_special_tokens_extended attribute — fixing both errors at once.
transformers>=4.51.0

# vllm 0.8+ has native Qwen3-VL support and works with transformers >=4.51
vllm>=0.8.0,<0.9.0
